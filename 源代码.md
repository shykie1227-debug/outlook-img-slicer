```python
class ProcessWorker(QThread):
    progress = Signal(int)
    finished = Signal(list)
    error = Signal(str)

    def __init__(self, file_path: str, width: int):
        super().__init__()
        self.file_path = file_path
        self.width = width

        try:
            ext = Path(self.file_path).suffix.lower()
            self.progress.emit(10)
            if ext == ".pdf":
                images = pdf_to_images(self.file_path)
                self.progress.emit(30)
                slice_paths = []
                temp_dir = tempfile.gettempdir()
                for i, img in enumerate(images):
                    path = os.path.join(temp_dir, f"pdf_page_{i}.png")
                    img.save(path)
                    slice_paths.append(path)
                self.progress.emit(60)
            else:
                slice_paths = detect_and_slice(self.file_path)
                self.progress.emit(60)
            self.progress.emit(100)
            self.finished.emit(slice_paths)
        except Exception as e:
            self.error.emit(str(e))
```

```python
"""
图像切片器模块
将长图按指定高度切片，支持 JPG/PNG/BMP/WebP/GIF
"""
import os
import tempfile
from typing import List
from pathlib import Path
from PIL import Image


def detect_and_slice(image_path: str, max_height: int = 1500) -> List[str]:
    """
    检测图像高度，必要时切片

    Args:
        image_path: 图片文件路径
        max_height: 单片最大高度，默认 1500px

    Returns:
        切片后的临时文件路径列表
    """
    img = Image.open(image_path)
    orig_w, orig_h = img.size

    # 无需切片
    if orig_h <= max_height:
        return [image_path]

    # 计算切片数量
    slice_count = (orig_h + max_height - 1) // max_height
    slice_paths: List[str] = []
    temp_dir = tempfile.gettempdir()

    for i in range(slice_count):
        top = i * max_height
        bottom = min((i + 1) * max_height, orig_h)
        slice_h = bottom - top

        # 裁剪对应区域，保持原始宽度不变
        slice_img = img.crop((0, top, orig_w, bottom))
        slice_img = slice_img.resize((orig_w, slice_h), Image.LANCZOS)

        # 保存切片
        ext = Path(image_path).suffix.lower()
        slice_path = os.path.join(temp_dir, f"slice_{i}{ext}")
        slice_img.save(slice_path, quality=95)
        slice_paths.append(slice_path)

    return slice_paths


def get_image_info(image_path: str) -> dict:
    """获取图片基本信息"""
    img = Image.open(image_path)
    w, h = img.size
    ext = Path(image_path).suffix.lower()
    return {"width": w, "height": h, "format": ext}
```

```python
"""
PDF 解析模块
使用 PyMuPDF 将 PDF 页面转换为 PIL Image
"""
from typing import List
import fitz  # PyMuPDF


def pdf_to_images(pdf_path: str, dpi: int = 150) -> List:
    """
    将 PDF 所有页面渲染为 PIL Image

    Args:
        pdf_path: PDF 文件路径
        dpi: 渲染分辨率，默认 150

    Returns:
        PIL Image 列表
    """
    images = []
    doc = fitz.open(pdf_path)
    for page in doc:
        # 计算缩放比例
        mat = fitz.Matrix(dpi / 72, dpi / 72)
        pix = page.get_pixmap(matrix=mat)
        img_data = pix.tobytes("png")
        from PIL import Image
        import io
        img = Image.open(io.BytesIO(img_data))
        images.append(img)
    doc.close()
    return images
```

## 4. html_assembler.py — HTML 表格生成

```python
"""
HTML 组装器模块
生成带内联 CSS 的表格 HTML，实现图片无缝居中拼接
"""
from typing import List


def assemble_html(image_paths: List[str], original_width: int = 1000) -> str:
    """
    生成 HTML 邮件内容（居中表格布局）

    Args:
        image_paths: 切片图片路径列表
        original_width: 原始图片宽度（px）

    Returns:
        HTML 字符串
    """
    img_tags = ""
    for path in image_paths:
        # 构建绝对路径的 file:// URL（供邮件使用）
        file_url = "file:///" + path.replace("\\", "/")
        img_tags += (
            f'<img src="{file_url}" '
            f'width="{original_width}" '
            f'style="'
            f'display: block; '
            f'margin: 0 auto; '
            f'width: {original_width}px; '
            f'" />'
        )

    html = (
        f'<div align="center">\n'
        f'<table cellpadding="0" cellspacing="0" border="0" '
        f'style="margin: 0 auto; border-collapse: collapse;">\n'
        f'<tr>\n'
        f'<td style="padding: 0;">\n'
        + img_tags +
        f'</td>\n'
        f'</tr>\n'
        f'</table>\n'
        f'</div>'
    )
    return html
```

```python
import sys


def create_email_with_images(html_content: str, subject: str = "",
                              to: str = "", attachments=None):
    """
    创建 Outlook 邮件窗口并填充内容

    Args:
        html_content: HTML 邮件正文
        subject: 邮件主题
        to: 收件人（可选）
        attachments: 附件路径列表（可选）
    """
    if sys.platform != "win32":
        raise RuntimeError("Outlook 自动化仅支持 Windows 系统")

    import win32com.client
    outlook = win32com.client.Dispatch("Outlook.Application")
    mail = outlook.CreateItem(0)  # 0 = olMailItem

    mail.HTMLBody = html_content
    if subject:
        mail.Subject = subject
    if to:
        mail.To = to

    if attachments:
        for path in attachments:
            mail.Attachments.Add(path)

    mail.Display()  # 显示邮件窗口（用户手动发送）
```
